1//! Path2D implementation for reusable path objects.
2//!
3//! Path2D allows creating path objects that can be reused across multiple
4//! fill, stroke, or clip operations.
5
6use crate::error::{Canvas2dError, Canvas2dResult};
7use crate::geometry::{
8    ArcParams, ArcToParams, CubicBezierParams, EllipseParams, QuadraticBezierParams, RectParams,
9    RoundRectParams,
10};
11use tiny_skia::PathBuilder;
12
13/// A reusable path object that can be used with fill, stroke, and clip operations.
14#[derive(Debug, Clone)]
15pub struct Path2D {
16    /// The underlying path builder for constructing the path.
17    builder: PathBuilder,
18    /// Cached finished path (invalidated when path is modified).
19    path: Option<tiny_skia::Path>,
20    /// Current position for tracking subpath.
21    current_x: f32,
22    current_y: f32,
23    /// Subpath start for closePath.
24    subpath_start_x: f32,
25    subpath_start_y: f32,
26    /// Whether the path has a current point (for arc line_to vs move_to).
27    has_current_point: bool,
28}
29
30impl Default for Path2D {
31    fn default() -> Self {
32        Self::new()
33    }
34}
35
36impl Path2D {
37    /// Create an empty path.
38    pub fn new() -> Self {
39        Self {
40            builder: PathBuilder::new(),
41            path: None,
42            current_x: 0.0,
43            current_y: 0.0,
44            subpath_start_x: 0.0,
45            subpath_start_y: 0.0,
46            has_current_point: false,
47        }
48    }
49
50    /// Create a copy of another Path2D.
51    pub fn from_path(other: &Path2D) -> Self {
52        other.clone()
53    }

55    /// Create a Path2D from SVG path data string.
56    ///
57    /// Supports all SVG path commands (M, L, H, V, Q, T, C, S, A, Z).
58    /// Arc commands are automatically converted to cubic Bezier curves.
59    ///
60    /// # Example
61    /// ```
62    /// use vl_convert_canvas2d::Path2D;
63    ///
64    /// let path = Path2D::from_svg_path_data("M10,10 L50,50 A10,10 0 0 1 100,100 Z").unwrap();
65    /// ```
66    pub fn from_svg_path_data(path_data: &str) -> Canvas2dResult<Self> {
67        let mut path = Path2D::new();
68
69        for segment in svgtypes::SimplifyingPathParser::from(path_data) {
70            let segment = segment.map_err(|e| {
71                Canvas2dError::InvalidArgument(format!("Invalid SVG path data: {:?}", e))
72            })?;
73
74            match segment {
75                svgtypes::SimplePathSegment::MoveTo { x, y } => {
76                    path.move_to(x as f32, y as f32);
77                }
78                svgtypes::SimplePathSegment::LineTo { x, y } => {
79                    path.line_to(x as f32, y as f32);
80                }
81                svgtypes::SimplePathSegment::Quadratic { x1, y1, x, y } => {
82                    path.quadratic_curve_to(&QuadraticBezierParams {
83                        cpx: x1 as f32,
84                        cpy: y1 as f32,
85                        x: x as f32,
86                        y: y as f32,
87                    });
88                }
89                svgtypes::SimplePathSegment::CurveTo {
90                    x1,
91                    y1,
92                    x2,
93                    y2,
94                    x,
95                    y,
96                } => {
97                    path.bezier_curve_to(&CubicBezierParams {
98                        cp1x: x1 as f32,
99                        cp1y: y1 as f32,
100                        cp2x: x2 as f32,
101                        cp2y: y2 as f32,
102                        x: x as f32,
103                        y: y as f32,
104                    });
105                }
106                svgtypes::SimplePathSegment::ClosePath => {
107                    path.close_path();
108                }
109            }
110        }
111
112        Ok(path)
113    }

115    /// Invalidate the cached path (called when path is modified).
116    fn invalidate(&mut self) {
117        self.path = None;
118    }

120    /// Move to a point without drawing.
121    pub fn move_to(&mut self, x: f32, y: f32) {
122        self.invalidate();
123        self.builder.move_to(x, y);
124        self.current_x = x;
125        self.current_y = y;
126        self.subpath_start_x = x;
127        self.subpath_start_y = y;
128        self.has_current_point = true;
129    }

131    /// Draw a line to a point.
132    pub fn line_to(&mut self, x: f32, y: f32) {
133        self.invalidate();
134        self.builder.line_to(x, y);
135        self.current_x = x;
136        self.current_y = y;
137        self.has_current_point = true;
138    }

140    /// Close the current subpath.
141    pub fn close_path(&mut self) {
142        self.invalidate();
143        self.builder.close();
144        self.current_x = self.subpath_start_x;
145        self.current_y = self.subpath_start_y;
146    }

148    /// Add a cubic bezier curve.
149    pub fn bezier_curve_to(&mut self, params: &CubicBezierParams) {
150        self.invalidate();
151        self.builder.cubic_to(
152            params.cp1x,
153            params.cp1y,
154            params.cp2x,
155            params.cp2y,
156            params.x,
157            params.y,
158        );
159        self.current_x = params.x;
160        self.current_y = params.y;
161    }

163    /// Add a quadratic bezier curve.
164    pub fn quadratic_curve_to(&mut self, params: &QuadraticBezierParams) {
165        self.invalidate();
166        self.builder
167            .quad_to(params.cpx, params.cpy, params.x, params.y);
168        self.current_x = params.x;
169        self.current_y = params.y;
170    }

172    /// Add a rectangle to the path.
173    pub fn rect(&mut self, params: &RectParams) {
174        self.invalidate();
175        self.move_to(params.x, params.y);
176        self.line_to(params.x + params.width, params.y);
177        self.line_to(params.x + params.width, params.y + params.height);
178        self.line_to(params.x, params.y + params.height);
179        self.close_path();
180    }

182    /// Add a rounded rectangle to the path.
183    pub fn round_rect(&mut self, params: &RoundRectParams) {
184        use crate::geometry::CornerRadius;
185        self.invalidate();
186
187        // Handle negative dimensions
188        let (x, width) = if params.width < 0.0 {
189            (params.x + params.width, -params.width)
190        } else {
191            (params.x, params.width)
192        };
193        let (y, height) = if params.height < 0.0 {
194            (params.y + params.height, -params.height)
195        } else {
196            (params.y, params.height)
197        };
198
199        let [mut tl, mut tr, mut br, mut bl] = params.radii;
200
201        // Clamp radii to non-negative
202        tl = CornerRadius {
203            x: tl.x.max(0.0),
204            y: tl.y.max(0.0),
205        };
206        tr = CornerRadius {
207            x: tr.x.max(0.0),
208            y: tr.y.max(0.0),
209        };
210        br = CornerRadius {
211            x: br.x.max(0.0),
212            y: br.y.max(0.0),
213        };
214        bl = CornerRadius {
215            x: bl.x.max(0.0),
216            y: bl.y.max(0.0),
217        };
218
219        // Scale radii uniformly if they exceed the rectangle dimensions
220        let top = (tl.x + tr.x).max(1e-10);
221        let bottom = (bl.x + br.x).max(1e-10);
222        let left = (tl.y + bl.y).max(1e-10);
223        let right = (tr.y + br.y).max(1e-10);
224        let scale = (width / top)
225            .min(width / bottom)
226            .min(height / left)
227            .min(height / right)
228            .min(1.0);
229
230        if scale < 1.0 {
231            tl.x *= scale;
232            tl.y *= scale;
233            tr.x *= scale;
234            tr.y *= scale;
235            br.x *= scale;
236            br.y *= scale;
237            bl.x *= scale;
238            bl.y *= scale;
239        }
240
241        // Kappa for quarter-ellipse cubic Bezier approximation
242        const K: f32 = 0.552_284_8;
243
244        // Build rounded rectangle path with elliptical corners
245        self.builder.move_to(x + tl.x, y);
246
247        // Top edge
248        self.builder.line_to(x + width - tr.x, y);
249
250        // Top-right corner
251        if tr.x > 0.0 || tr.y > 0.0 {
252            self.builder.cubic_to(
253                x + width - tr.x + tr.x * K,
254                y,
255                x + width,
256                y + tr.y - tr.y * K,
257                x + width,
258                y + tr.y,
259            );
260        }
261
262        // Right edge
263        self.builder.line_to(x + width, y + height - br.y);
264
265        // Bottom-right corner
266        if br.x > 0.0 || br.y > 0.0 {
267            self.builder.cubic_to(
268                x + width,
269                y + height - br.y + br.y * K,
270                x + width - br.x + br.x * K,
271                y + height,
272                x + width - br.x,
273                y + height,
274            );
275        }
276
277        // Bottom edge
278        self.builder.line_to(x + bl.x, y + height);
279
280        // Bottom-left corner
281        if bl.x > 0.0 || bl.y > 0.0 {
282            self.builder.cubic_to(
283                x + bl.x - bl.x * K,
284                y + height,
285                x,
286                y + height - bl.y + bl.y * K,
287                x,
288                y + height - bl.y,
289            );
290        }
291
292        // Left edge
293        self.builder.line_to(x, y + tl.y);
294
295        // Top-left corner
296        if tl.x > 0.0 || tl.y > 0.0 {
297            self.builder
298                .cubic_to(x, y + tl.y - tl.y * K, x + tl.x - tl.x * K, y, x + tl.x, y);
299        }
300
301        self.builder.close();
302
303        self.subpath_start_x = x + tl.x;
304        self.subpath_start_y = y;
305        self.current_x = x + tl.x;
306        self.current_y = y;
307    }

329    /// Add the segments of another path to this path, optionally applying a transform.
330    pub fn add_path(
331        &mut self,
332        other: &mut Path2D,
333        transform: Option<crate::dom_matrix::DOMMatrix>,
334    ) {
335        let Some(other_path) = other.get_path() else {
336            return;
337        };
338
339        let ts: tiny_skia::Transform = transform
340            .map(|m| m.into())
341            .unwrap_or(tiny_skia::Transform::identity());
342
343        let map = |x: f32, y: f32| -> (f32, f32) {
344            (ts.sx * x + ts.kx * y + ts.tx, ts.ky * x + ts.sy * y + ts.ty)
345        };
346
347        self.invalidate();
348
349        for seg in other_path.segments() {
350            match seg {
351                tiny_skia::PathSegment::MoveTo(pt) => {
352                    let (x, y) = map(pt.x, pt.y);
353                    self.builder.move_to(x, y);
354                    self.current_x = x;
355                    self.current_y = y;
356                    self.subpath_start_x = x;
357                    self.subpath_start_y = y;
358                }
359                tiny_skia::PathSegment::LineTo(pt) => {
360                    let (x, y) = map(pt.x, pt.y);
361                    self.builder.line_to(x, y);
362                    self.current_x = x;
363                    self.current_y = y;
364                }
365                tiny_skia::PathSegment::QuadTo(pt1, pt2) => {
366                    let (x1, y1) = map(pt1.x, pt1.y);
367                    let (x2, y2) = map(pt2.x, pt2.y);
368                    self.builder.quad_to(x1, y1, x2, y2);
369                    self.current_x = x2;
370                    self.current_y = y2;
371                }
372                tiny_skia::PathSegment::CubicTo(pt1, pt2, pt3) => {
373                    let (x1, y1) = map(pt1.x, pt1.y);
374                    let (x2, y2) = map(pt2.x, pt2.y);
375                    let (x3, y3) = map(pt3.x, pt3.y);
376                    self.builder.cubic_to(x1, y1, x2, y2, x3, y3);
377                    self.current_x = x3;
378                    self.current_y = y3;
379                }
380                tiny_skia::PathSegment::Close => {
381                    self.builder.close();
382                    self.current_x = self.subpath_start_x;
383                    self.current_y = self.subpath_start_y;
384                }
385            }
386        }
387        self.has_current_point = true;
388    }

390    /// Get the finished path for rendering.
391    /// Returns None if the path is empty.
392    pub(crate) fn get_path(&mut self) -> Option<&tiny_skia::Path> {
393        if self.path.is_none() {
394            // Clone the builder to finish it without consuming it
395            let builder_clone = self.builder.clone();
396            self.path = builder_clone.finish();
397        }
398        self.path.as_ref()
399    }
