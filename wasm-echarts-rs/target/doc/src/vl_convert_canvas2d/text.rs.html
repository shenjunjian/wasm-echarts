8/// Text metrics returned by measureText().
9#[derive(Debug, Clone, Default)]
10pub struct TextMetrics {
11    /// Width of the text in pixels.
12    pub width: f32,
13    /// Distance from baseline to top of the bounding box.
14    pub actual_bounding_box_ascent: f32,
15    /// Distance from baseline to bottom of the bounding box.
16    pub actual_bounding_box_descent: f32,
17    /// Font ascent.
18    pub font_bounding_box_ascent: f32,
19    /// Font descent.
20    pub font_bounding_box_descent: f32,
21    /// Distance from alignment point to left of the bounding box.
22    pub actual_bounding_box_left: f32,
23    /// Distance from alignment point to right of the bounding box.
24    pub actual_bounding_box_right: f32,
25}

27/// Result of resolving a font family by post_script_name.
28/// Includes both the actual family name and the font weight from the matched face.
29struct PostScriptNameMatch {
30    family_name: String,
31    weight: Weight,
32}
33
34/// Try to resolve a font family name by checking post_script_names in the fontdb.
35/// This handles cases like "Matter SemiBold" where the CSS font name doesn't match
36/// the font family name but matches the PostScript name (e.g., "Matter-SemiBold").
37///
38/// Returns the actual font family name and weight if found via post_script_name lookup,
39/// or None if no match is found.
40fn resolve_family_by_postscript_name(
41    font_system: &FontSystem,
42    family: &str,
43) -> Option<PostScriptNameMatch> {
44    let normalized_name = family.replace('-', " ");
45
46    for face in font_system.db().faces() {
47        let post_script_normalized = face.post_script_name.replace('-', " ");
48        if post_script_normalized.eq_ignore_ascii_case(&normalized_name) {
49            // Found a match - return the actual family name and weight from this face
50            if let Some((family_name, _)) = face.families.first() {
51                return Some(PostScriptNameMatch {
52                    family_name: family_name.clone(),
53                    weight: Weight(face.weight.0),
54                });
55            }
56        }
57    }
58
59    None
60}

62/// Result of family resolution with optional weight override.
63pub struct FamilyResolution<'a> {
64    pub family: Family<'a>,
65    pub weight_override: Option<Weight>,
66}
67
68/// Get the Family enum for a font, with post_script_name fallback lookup.
69/// When a font is found via post_script_name, also returns the font's actual weight
70/// to override any weight parsed from the CSS font string (e.g., "bold 13px Matter SemiBold"
71/// should use weight 600 from "Matter-SemiBold", not weight 700 from "bold").
72pub fn get_family_with_fallback<'a>(
73    font_system: &FontSystem,
74    family: &'a str,
75    resolved_name: &'a mut Option<String>,
76) -> FamilyResolution<'a> {
77    // First check for generic family names
78    match family.to_lowercase().as_str() {
79        "sans-serif" => {
80            return FamilyResolution {
81                family: Family::SansSerif,
82                weight_override: None,
83            }
84        }
85        "serif" => {
86            return FamilyResolution {
87                family: Family::Serif,
88                weight_override: None,
89            }
90        }
91        "monospace" => {
92            return FamilyResolution {
93                family: Family::Monospace,
94                weight_override: None,
95            }
96        }
97        "cursive" => {
98            return FamilyResolution {
99                family: Family::Cursive,
100                weight_override: None,
101            }
102        }
103        "fantasy" => {
104            return FamilyResolution {
105                family: Family::Fantasy,
106                weight_override: None,
107            }
108        }
109        _ => {}
110    }
111
112    // Try post_script_name fallback lookup
113    if let Some(matched) = resolve_family_by_postscript_name(font_system, family) {
114        *resolved_name = Some(matched.family_name);
115        // Return reference to the resolved name stored in the Option
116        // Also return the weight from the matched font face
117        return FamilyResolution {
118            family: Family::Name(resolved_name.as_ref().unwrap()),
119            weight_override: Some(matched.weight),
120        };
121    }
122
123    // Use the original family name
124    FamilyResolution {
125        family: Family::Name(family),
126        weight_override: None,
127    }
128}

130/// Measure text using cosmic-text.
131pub fn measure_text(
132    font_system: &mut FontSystem,
133    text: &str,
134    font: &ParsedFont,
135) -> Canvas2dResult<TextMetrics> {
136    let metrics = Metrics::new(font.size_px, font.size_px * 1.2);
137    let mut buffer = Buffer::new(font_system, metrics);
138
139    // Build attributes from parsed font
140    // Convert generic family names to proper Family enum variants,
141    // with fallback lookup by post_script_name for custom fonts like "Matter SemiBold"
142    let mut resolved_name: Option<String> = None;
143    let resolution = font
144        .families
145        .first()
146        .map(|f| get_family_with_fallback(font_system, f, &mut resolved_name))
147        .unwrap_or(FamilyResolution {
148            family: Family::SansSerif,
149            weight_override: None,
150        });
151
152    // Use weight from post_script_name match if available, otherwise use parsed CSS weight.
153    // This handles cases like "bold 13px Matter SemiBold" where the CSS "bold" (700) should
154    // be overridden by the actual font weight (600) from "Matter-SemiBold".
155    let weight = resolution.weight_override.unwrap_or(font.weight);
156
157    let attrs = Attrs::new()
158        .family(resolution.family)
159        .weight(weight)
160        .style(font.style)
161        .stretch(font.stretch.into());
162
163    buffer.set_text(font_system, text, &attrs, Shaping::Advanced, None);
164    buffer.shape_until_scroll(font_system, false);
165
166    // Calculate width from layout runs
167    let mut width: f32 = 0.0;
168    for run in buffer.layout_runs() {
169        width = width.max(run.line_w);
170    }
171
172    // Derive ascent/descent from shaped layout output for better browser parity.
173    let mut font_ascent: f32 = 0.0;
174    let mut font_descent: f32 = 0.0;
175    for line in &buffer.lines {
176        if let Some(layout_lines) = line.layout_opt() {
177            for layout_line in layout_lines {
178                font_ascent = font_ascent.max(layout_line.max_ascent);
179                font_descent = font_descent.max(layout_line.max_descent);
180            }
181        }
182    }
183    if font_ascent == 0.0 && font_descent == 0.0 {
184        font_ascent = font.size_px * 0.8;
185        font_descent = font.size_px * 0.2;
186    }
187
188    Ok(TextMetrics {
189        width,
190        actual_bounding_box_ascent: font_ascent,
191        actual_bounding_box_descent: font_descent,
192        font_bounding_box_ascent: font_ascent,
193        font_bounding_box_descent: font_descent,
194        actual_bounding_box_left: 0.0,
195        actual_bounding_box_right: width,
196    })
197}

199/// Calculate X offset for text alignment.
200pub fn calculate_text_x_offset(width: f32, align: TextAlign) -> f32 {
201    match align {
202        TextAlign::Left | TextAlign::Start => 0.0,
203        TextAlign::Right | TextAlign::End => -width,
204        TextAlign::Center => -width / 2.0,
205    }
206}
207
208/// Calculate Y offset for text baseline.
209pub fn calculate_text_y_offset(ascent: f32, descent: f32, baseline: TextBaseline) -> f32 {
210    match baseline {
211        TextBaseline::Top => ascent,
212        TextBaseline::Hanging => ascent * 0.8,
213        TextBaseline::Middle => ascent / 2.0 - descent / 2.0,
214        TextBaseline::Alphabetic => 0.0,
215        TextBaseline::Ideographic => -descent * 0.5,
216        TextBaseline::Bottom => -descent,
217    }
218}
