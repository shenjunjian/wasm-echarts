126/// Font stretch (width) for text rendering.
127///
128/// Maps to CSS `font-stretch` keywords and `cosmic_text::Stretch`.
129#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
130pub enum FontStretch {
131    /// Ultra-condensed width.
132    UltraCondensed,
133    /// Extra-condensed width.
134    ExtraCondensed,
135    /// Condensed width.
136    Condensed,
137    /// Semi-condensed width.
138    SemiCondensed,
139    /// Normal width.
140    #[default]
141    Normal,
142    /// Semi-expanded width.
143    SemiExpanded,
144    /// Expanded width.
145    Expanded,
146    /// Extra-expanded width.
147    ExtraExpanded,
148    /// Ultra-expanded width.
149    UltraExpanded,
150}
151
152impl FontStretch {
153    /// Convert from a CSS keyword string. Returns None for invalid values.
154    pub fn from_css_keyword(s: &str) -> Option<Self> {
155        match s {
156            "ultra-condensed" => Some(Self::UltraCondensed),
157            "extra-condensed" => Some(Self::ExtraCondensed),
158            "condensed" => Some(Self::Condensed),
159            "semi-condensed" => Some(Self::SemiCondensed),
160            "normal" => Some(Self::Normal),
161            "semi-expanded" => Some(Self::SemiExpanded),
162            "expanded" => Some(Self::Expanded),
163            "extra-expanded" => Some(Self::ExtraExpanded),
164            "ultra-expanded" => Some(Self::UltraExpanded),
165            _ => None,
166        }
167    }
168
169    /// Convert to CSS keyword string.
170    pub fn as_css_keyword(&self) -> &'static str {
171        match self {
172            Self::UltraCondensed => "ultra-condensed",
173            Self::ExtraCondensed => "extra-condensed",
174            Self::Condensed => "condensed",
175            Self::SemiCondensed => "semi-condensed",
176            Self::Normal => "normal",
177            Self::SemiExpanded => "semi-expanded",
178            Self::Expanded => "expanded",
179            Self::ExtraExpanded => "extra-expanded",
180            Self::UltraExpanded => "ultra-expanded",
181        }
182    }
183}
