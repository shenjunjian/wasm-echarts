53    /// Fill text at the specified position.

55        log::debug!(target: "canvas", "fillText \"{}\" {} {}", text, x, y);
56        self.render_text_impl(text, x, y, None, true);

81    /// Internal text rendering using vector glyph paths (used by fillText and strokeText).
82    fn render_text_impl(&mut self, text: &str, x: f32, y: f32, max_width: Option<f32>, fill: bool) {
83        // Handle max_width edge cases: if <= 0 or NaN, don't render
84        if let Some(mw) = max_width {
85            if mw <= 0.0 || mw.is_nan() {
86                return;
87            }
88        }
89
90        let font = &self.state.font;
91        let metrics = Metrics::new(font.size_px, font.size_px * 1.2);
92        let mut buffer = Buffer::new(&mut self.font_system, metrics);
93
94        // Build attributes from parsed font
95        // Use get_family_with_fallback for proper font resolution (same as measure_text)
96        let mut resolved_name: Option<String> = None;
97        let resolution = font
98            .families
99            .first()
100            .map(|f| {
101                crate::text::get_family_with_fallback(&self.font_system, f, &mut resolved_name)
102            })
103            .unwrap_or(crate::text::FamilyResolution {
104                family: Family::SansSerif,
105                weight_override: None,
106            });
107
108        // Use weight from post_script_name match if available, otherwise use parsed CSS weight
109        let weight = resolution.weight_override.unwrap_or(font.weight);
110
111        // Build attributes including letter spacing if set
112        let letter_spacing = self.state.letter_spacing;
113        let cache_key_flags = if self.hinting_enabled {
114            CacheKeyFlags::empty()
115        } else {
116            CacheKeyFlags::DISABLE_HINTING
117        };
118        let attrs = Attrs::new()
119            .family(resolution.family)
120            .weight(weight)
121            .style(font.style)
122            .stretch(font.stretch.into())
123            .letter_spacing(letter_spacing)
124            .cache_key_flags(cache_key_flags);
125
126        buffer.set_text(&mut self.font_system, text, &attrs, Shaping::Advanced, None);
127        buffer.shape_until_scroll(&mut self.font_system, false);
128
129        // Get text dimensions for alignment
130        let mut text_width: f32 = 0.0;
131        let mut text_ascent: f32 = 0.0;
132        let mut text_descent: f32 = 0.0;
133        for run in buffer.layout_runs() {
134            text_width = text_width.max(run.line_w);
135            text_ascent = text_ascent.max(run.line_y - run.line_top);
136            text_descent = text_descent.max((run.line_top + run.line_height) - run.line_y);
137        }
138        if text_ascent == 0.0 && text_descent == 0.0 {
139            text_ascent = font.size_px * 0.8;
140            text_descent = font.size_px * 0.2;
141        }
142
143        // Calculate horizontal scale factor for maxWidth
144        let scale_x = if let Some(mw) = max_width {
145            if mw.is_infinite() || text_width <= mw {
146                // Infinity or text fits: no scaling needed
147                1.0
148            } else {
149                // Text is too wide: calculate scale factor
150                let scale = mw / text_width;
151                // Don't render if scale would be too small (< 0.1%)
152                if scale < 0.001 {
153                    return;
154                }
155                scale
156            }
157        } else {
158            1.0
159        };
160
161        // Calculate alignment offset using ORIGINAL text width.
162        // The scale transform (applied around x) will handle making the scaled text
163        // properly aligned - if we used scaled_text_width here, we'd double-adjust.
164        let x_offset = crate::text::calculate_text_x_offset(text_width, self.state.text_align);
165
166        // Calculate baseline offset
167        let y_offset = crate::text::calculate_text_y_offset(
168            text_ascent,
169            text_descent,
170            self.state.text_baseline,
171        );
172
173        // Calculate base position with alignment offsets
174        // Note: We use (x, y) as the anchor point, x_offset adjusts for alignment
175        let base_x = x + x_offset;
176        let base_y = y + y_offset;
177
178        // Get the current transform
179        let transform = self.state.transform;
180
181        // For maxWidth scaling, we need to scale around the text anchor point (x position).
182        // Build a combined transform that:
183        // 1. Translates to put the anchor at origin
184        // 2. Scales horizontally
185        // 3. Translates back
186        // 4. Applies global transform
187        let scale_transform = if scale_x != 1.0 {
188            // Scale around the x anchor point (keeping y unchanged)
189            Transform::from_translate(x, 0.0)
190                .pre_scale(scale_x, 1.0)
191                .pre_translate(-x, 0.0)
192                .post_concat(transform)
193        } else {
194            transform
195        };
196
197        // Get the paint for rendering text and render while it's alive.
198        let style = if fill {
199            self.state.fill_style.clone()
200        } else {
201            self.state.stroke_style.clone()
202        };
203        let _ = self.with_paint_from_style(style, |ctx, paint| {
204            // Render each glyph as a vector path
205            for run in buffer.layout_runs() {
206                for glyph in run.glyphs.iter() {
207                    // Get the cache key for outline retrieval (physical() provides this)
208                    let physical_glyph = glyph.physical((base_x, base_y), 1.0);
209
210                    // Calculate floating-point glyph position for sub-pixel precision
211                    // (matching how usvg/resvg positions glyphs)
212                    let glyph_x = base_x + glyph.x + glyph.font_size * glyph.x_offset;
213                    let glyph_y = base_y + glyph.y - glyph.font_size * glyph.y_offset;
214
215                    // Get outline commands for this glyph
216                    if let Some(commands) = ctx
217                        .swash_cache
218                        .get_outline_commands(&mut ctx.font_system, physical_glyph.cache_key)
219                    {
220                        // Build a path from the outline commands
221                        // Note: Font outlines have Y pointing up, screen has Y pointing down
222                        // so we negate Y coordinates during path building
223                        let mut path_builder = tiny_skia::PathBuilder::new();
224                        for cmd in commands {
225                            match cmd {
226                                Command::MoveTo(p) => path_builder.move_to(p.x, -p.y),
227                                Command::LineTo(p) => path_builder.line_to(p.x, -p.y),
228                                Command::QuadTo(ctrl, end) => {
229                                    path_builder.quad_to(ctrl.x, -ctrl.y, end.x, -end.y)
230                                }
231                                Command::CurveTo(c1, c2, end) => {
232                                    path_builder.cubic_to(c1.x, -c1.y, c2.x, -c2.y, end.x, -end.y)
233                                }
234                                Command::Close => path_builder.close(),
235                            }
236                        }
237
238                        if let Some(path) = path_builder.finish() {
239                            // Create a transform that positions the glyph correctly
240                            // Using floating-point position for sub-pixel precision
241                            let glyph_transform = Transform::from_translate(glyph_x, glyph_y)
242                                .post_concat(scale_transform);
243
244                            if fill {
245                                // Fill the glyph path
246                                ctx.pixmap.fill_path(
247                                    &path,
248                                    paint,
249                                    tiny_skia::FillRule::Winding,
250                                    glyph_transform,
251                                    None,
252                                );
253                            } else {
254                                // Stroke the glyph path
255                                let stroke = tiny_skia::Stroke {
256                                    width: ctx.state.line_width,
257                                    line_cap: ctx.state.line_cap.into(),
258                                    line_join: ctx.state.line_join.into(),
259                                    miter_limit: ctx.state.miter_limit,
260                                    dash: if ctx.state.line_dash.is_empty() {
261                                        None
262                                    } else {
263                                        tiny_skia::StrokeDash::new(
264                                            ctx.state.line_dash.clone(),
265                                            ctx.state.line_dash_offset,
266                                        )
267                                    },
268                                };
269                                ctx.pixmap.stroke_path(
270                                    &path,
271                                    paint,
272                                    &stroke,
273                                    glyph_transform,
274                                    None,
275                                );
276                            }
277                        }
278                    }
279                }
280            }
281        });
282    }
