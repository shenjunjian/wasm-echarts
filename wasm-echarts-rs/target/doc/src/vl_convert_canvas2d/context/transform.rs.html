1//! Transform operations for Canvas2dContext.
2
3use super::Canvas2dContext;
4use crate::dom_matrix::DOMMatrix;
5use tiny_skia::Transform;
6
7impl Canvas2dContext {
8    /// Translate the canvas.
9    pub fn translate(&mut self, x: f32, y: f32) {
10        log::debug!(target: "canvas", "translate {} {}", x, y);
11        self.state.transform = self.state.transform.pre_translate(x, y);
12    }
13
14    /// Rotate the canvas.
15    pub fn rotate(&mut self, angle: f32) {
16        log::debug!(target: "canvas", "rotate {}", angle);
17        let cos = angle.cos();
18        let sin = angle.sin();
19        let rotation = Transform::from_row(cos, sin, -sin, cos, 0.0, 0.0);
20        self.state.transform = self.state.transform.pre_concat(rotation);
21    }
22
23    /// Scale the canvas.
24    pub fn scale(&mut self, x: f32, y: f32) {
25        log::debug!(target: "canvas", "scale {} {}", x, y);
26        self.state.transform = self.state.transform.pre_scale(x, y);
27    }
28
29    /// Apply a transform matrix.
30    pub fn transform(&mut self, matrix: DOMMatrix) {
31        log::debug!(target: "canvas", "transform {:?}", matrix);
32        let t: Transform = matrix.into();
33        self.state.transform = self.state.transform.pre_concat(t);
34    }
35
36    /// Set the transform matrix (replacing the current one).
37    pub fn set_transform(&mut self, matrix: DOMMatrix) {
38        log::debug!(target: "canvas", "setTransform {:?}", matrix);
39        self.state.transform = matrix.into();
40    }
41
42    /// Reset the transform to identity.
43    pub fn reset_transform(&mut self) {
44        log::debug!(target: "canvas", "resetTransform");
45        self.state.transform = Transform::identity();
46    }
47
48    /// Get the current transformation matrix.
49    pub fn get_transform(&self) -> DOMMatrix {
50        self.state.transform.into()
51    }
52}
