98    fn new_internal(
99        width: u32,
100        height: u32,
101        font_db: fontdb::Database,
102        hinting_enabled: bool,
103    ) -> Canvas2dResult<Self> {
104        // Validate dimensions
105        if width == 0 || height == 0 || width > MAX_DIMENSION || height > MAX_DIMENSION {
106            return Err(Canvas2dError::InvalidDimensions { width, height });
107        }
108
109        // Create pixmap
110        let pixmap =
111            Pixmap::new(width, height).ok_or(Canvas2dError::InvalidDimensions { width, height })?;
112
113        // Create font system from the provided (already-configured) fontdb
114        let font_system = FontSystem::new_with_locale_and_db("en".to_string(), font_db);
115
116        // Create swash cache for glyph rasterization
117        let swash_cache = SwashCache::new();
118
119        Ok(Self {
120            width,
121            height,
122            pixmap,
123            font_system,
124            swash_cache,
125            state: DrawingState::default(),
126            state_stack: Vec::new(),
127            clip_fill_rule: CanvasFillRule::NonZero,
128            clip_fill_rule_stack: Vec::new(),
129            path_builder: tiny_skia::PathBuilder::new(),
130            current_x: 0.0,
131            current_y: 0.0,
132            subpath_start_x: 0.0,
133            subpath_start_y: 0.0,
134            has_current_point: false,
135            pattern_pixmap_cache: PatternPixmapCache::new(PATTERN_CACHE_MAX_BYTES),
136            hinting_enabled,
137        })
138    }

162    /// Save the current drawing state.
163    pub fn save(&mut self) {
164        log::debug!(target: "canvas", "save");
165        self.state_stack.push(self.state.clone());
166        self.clip_fill_rule_stack.push(self.clip_fill_rule);
167    }

169    /// Restore the previously saved drawing state.
170    pub fn restore(&mut self) {
171        log::debug!(target: "canvas", "restore");
172        if let Some(state) = self.state_stack.pop() {
173            self.state = state;
174            self.clip_fill_rule = self
175                .clip_fill_rule_stack
176                .pop()
177                .unwrap_or(CanvasFillRule::NonZero);
178        }
179    }

181    /// Reset the rendering context to its default state.
182    ///
183    /// This clears the canvas to transparent, resets all drawing state
184    /// (fill/stroke style, transforms, etc.), and empties the state stack.
185    pub fn reset(&mut self) {
186        // Clear the canvas to transparent
187        self.pixmap.fill(tiny_skia::Color::TRANSPARENT);
188
189        // Reset state to defaults
190        self.state = DrawingState::default();
191
192        // Clear saved states
193        self.state_stack.clear();
194        self.clip_fill_rule_stack.clear();
195
196        // Reset path
197        self.path_builder = tiny_skia::PathBuilder::new();
198        self.current_x = 0.0;
199        self.current_y = 0.0;
200        self.subpath_start_x = 0.0;
201        self.subpath_start_y = 0.0;
202        self.has_current_point = false;
203        self.clip_fill_rule = CanvasFillRule::NonZero;
204
205        // Reset pattern cache so stale pattern backing pixmaps are released.
206        self.pattern_pixmap_cache.clear();
207    }

306    /// Set the line dash pattern.
307    /// Per spec: ignore if any value is non-finite or negative.
308    /// Duplicate odd-length arrays to make them even.
309    pub fn set_line_dash(&mut self, mut segments: Vec<f32>) {
310        // Reject if any value is non-finite or negative
311        if segments.iter().any(|&v| !v.is_finite() || v < 0.0) {
312            return;
313        }
314        // Duplicate odd-length arrays per spec
315        if !segments.len().is_multiple_of(2) {
316            let copy = segments.clone();
317            segments.extend(copy);
318        }
319        self.state.line_dash = segments;
320    }

444/// Parse a CSS color string into a tiny_skia::Color.
445pub(crate) fn parse_color(s: &str) -> Canvas2dResult<tiny_skia::Color> {
446    let parsed = csscolorparser::parse(s)
447        .map_err(|e| Canvas2dError::ColorParseError(format!("{}: {}", s, e)))?;
448
449    let [r, g, b, a] = parsed.to_array();
450    Ok(tiny_skia::Color::from_rgba(r, g, b, a).unwrap_or(tiny_skia::Color::BLACK))
451}

786

788        let pattern_data = vec![255_u8; 8 * 8 * 4];
789        let pattern = ctx.create_pattern(&pattern_data, 8, 8, "repeat").unwrap();
